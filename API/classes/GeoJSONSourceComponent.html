    mgl-geojson-source - a geojson source component

    geojson

    Example

    ...
    <mgl-map ...>
      <mgl-geojson-source id="symbols-source">
        <mgl-feature
          *ngFor="let geometry of geometries"
          [geometry]="geometry"
        ></mgl-feature>
      </mgl-geojson-source>
      ...
      <mgl-geojson-source
        id="earthquakes"
        [data]="earthquakes"
        [cluster]="true"
        [clusterMaxZoom]="14"
        [clusterRadius]="50"
      ></mgl-geojson-source>
    </mgl-map>
